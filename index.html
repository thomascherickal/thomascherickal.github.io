// Rust: async SLM agent via openfang + tokio
use openfang::{agent::{Agent, Config}, error::Result};
async fn run_agent(model: &str, task: &str) -> Result<String> {
    let cfg = Config {
        model: model.into(),
        base_url: "http://localhost:1234/v1".into(),
        stream: true, temperature: 0.2,
        ..Default::default()
    };
    Agent::new(cfg).await?.run(task).await
}
#[tokio::main]
async fn main() {
    let out = run_agent("qwen", "Compute 17+24").await.unwrap();
    println!("{}", out);
}

# Python: Generate text embeddings via OpenAI API
import os
from openai import OpenAI

def get_embedding(text: str, model="text-embedding-3-small"):
    client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
    try:
        resp = client.embeddings.create(input=[text], model=model)
        return resp.data[0].embedding
    except Exception as e:
        print(f"Error: {e}")
        return []

if __name__ == "__main__":
    vec = get_embedding("AI Systems")
    print(f"Dim: {len(vec)}")
genai_stream.go 
// Go: stream Gemini 2.0 — google/generative-ai-go
package main
import ("context"; "fmt"; "log"; "os"; "google.golang.org/api/option"
    genai "github.com/google/generative-ai-go/genai")
func main() {
    ctx := context.Background()
    c, err := genai.NewClient(ctx, option.WithAPIKey(os.Getenv("GEMINI_API_KEY")))
    if err != nil { log.Fatal(err) }
    defer c.Close()
    iter := c.GenerativeModel("gemini-2.0-flash").GenerateContentStream(ctx, genai.Text("Explain attention"))
    for {
        r, err := iter.Next()
        if err != nil { break }
        for _, p := range r.Candidates[0].Content.Parts { fmt.Print(p) }
    }
}
